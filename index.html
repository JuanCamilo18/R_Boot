2. Implementar una funcin que muestre el resultado de la ecuacin de Bhaskara y probar con valores.

Creamos la funcion ecu_Bhaskara que nos devuelva las soluciones para la ecuacin y ademas realice una grafica.

# ecuacin de Bhaskara
ecu_Bhaskara <- function(a,b,c){
  ecuacin <- paste0(a,'*x^2 + ', b,'*x + ', c, collapse = ",")
  #print(paste0('Resolviendo la ecuacion: ', a,'*x^2 + ', b,'*x +', c, collapse = ", "))
  print(paste0('Resolviendo la ecuacion: ', ecuacin, collapse = ", "))
  
  discriminante <- (b^2-4*a*c)
  
  if (discriminante < 0) {
    print("No existen soluciones Reales a la ecuacin!")
    x1 = (-b + sqrt(as.complex(b^2 - 4*a*c)))/(2*a)
    x2 = (-b - sqrt(as.complex(b^2 - 4*a*c)))/(2*a)
    }
  else {
    x1 = (-b + sqrt(discriminante))/(2*a)
    x2 = (-b - sqrt(discriminante))/(2*a)
    }
  cat(rep("-----",10),'\n')
  print('Las soluciones de la ecuacin:')
  print(paste0('La solucion x1 es: ', x1, ' y de x2 es: ' , x2, 
               collapse = ','))
  #Grafica de la ecuacin
  cat(rep("-----",10),"\nGrafica de la ecuacion: (", ecuacin, ") -> v ")
  curve(a*x^2 + b*x + c, xlim = c(-20,20), ylim = c(-10, 120), 
        col = "red", lwd=3.5)
  abline(h=0, v=0, col="blue", lwd=2.5)
}

ggplot(data.frame(observaciones), aes(observaciones))+
  geom_histogram(aes(y = ..density..),
                 colour = "white", fill = "red") +
  geom_density(lwd = 1.2, colour = "blue",
               fill = 4, alpha = 0.45) +
  ggtitle("Distribucion y densidad de observaciones") +
  theme(plot.title = element_text(hjust = 0.5, vjust = 1))
